    <script>

        const wrapText = (text) => {

            const maxLength = 16;

            if (text.length <= maxLength) {

                return text;

            }

            const words = text.split(' ');

            let lines = [];

            let currentLine = '';

            words.forEach(word => {

                if ((currentLine + ' ' + word).trim().length > maxLength) {

                    lines.push(currentLine.trim());

                    currentLine = word;

                } else {

                    currentLine = (currentLine + ' ' + word).trim();

                }

            });

            if (currentLine) {

                lines.push(currentLine.trim());

            }

            return lines;

        };


        const tooltipTitleCallback = (tooltipItems) => {

            const item = tooltipItems[0];

            let label = item.chart.data.labels[item.dataIndex];

            if (Array.isArray(label)) {

                return label.join(' ');

            }

            return label;

        };

        

        const tooltipPlugin = {

            tooltip: {

                callbacks: {

                    title: tooltipTitleCallback

                }

            }

        };

        const generateIdeasBtn = document.getElementById('generateIdeasBtn');

        const researchInterestInput = document.getElementById('researchInterest');

        const ideasOutput = document.getElementById('ideasOutput');

        const loadingSpinner = document.getElementById('loadingSpinner');


        generateIdeasBtn.addEventListener('click', async () => {

            const prompt = researchInterestInput.value.trim();

            if (!prompt) {

                ideasOutput.innerHTML = '<p class="text-red-500">Por favor, describe tus intereses para generar ideas.</p>';

                return;

            }


            ideasOutput.innerHTML = '';

            loadingSpinner.classList.remove('hidden');

            generateIdeasBtn.disabled = true;


            let chatHistory = [];

            chatHistory.push({ role: "user", parts: [{ text: `Genera 5 ideas de investigación innovadoras y aplicadas en el campo del diseño, la innovación y la experiencia de usuario, basadas en el siguiente interés: "${prompt}". Las ideas deben ser específicas y formuladas como posibles títulos de proyectos o preguntas de investigación.` }] });

            const payload = { contents: chatHistory };

            const apiKey = "";

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;


            try {

                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });

                const result = await response.json();


                if (result.candidates && result.candidates.length > 0 &&

                    result.candidates[0].content && result.candidates[0].content.parts &&

                    result.candidates[0].content.parts.length > 0) {

                    const text = result.candidates[0].content.parts[0].text;

                    ideasOutput.innerHTML = `<p class="text-gray-800 whitespace-pre-wrap">${text}</p>`;

                } else {

                    ideasOutput.innerHTML = '<p class="text-red-500">No se pudieron generar ideas. Intenta de nuevo.</p>';

                }

            } catch (error) {

                ideasOutput.innerHTML = `<p class="text-red-500">Error al conectar con la IA: ${error.message}.</p>`;

            } finally {

                loadingSpinner.classList.add('hidden');

                generateIdeasBtn.disabled = false;

            }

        });
